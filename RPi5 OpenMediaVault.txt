#!/bin/sh
# root-switch.sh — production-ready root switch for Pi 5 with SD boot + SATA root
# Author: ChatGPT
# Date: 2026-01-04

set -e

### CONFIGURATION ###
PARTUUID="9e020a5c-02"   # <-- replace with your SATA root PARTUUID
TIMEOUT=15               # seconds to wait for PCIe SATA
LOGFILE="/var/log/root-switch.log"
NEWROOT="/newroot"

### LOGGING FUNCTION ###
log() {
    msg="$(printf '%s [root-switch] %s\n' "$(date '+%Y-%m-%d %H:%M:%S')" "$*")"
    printf '%s\n' "$msg" > /dev/serial0
    printf '%s\n' "$msg" > /dev/console
}

log "Starting root-switch"

ROOTDEV=""

### WAIT FOR SATA ROOT ###
for i in $(seq 1 $TIMEOUT); do
    ROOTDEV=$(blkid -t PARTUUID="$PARTUUID" -o device 2>/dev/null || true)
    if [ -n "$ROOTDEV" ]; then
        # print a newline so the next message starts on a fresh line
        printf '\n' > /dev/serial0
        printf '\n' > /dev/console
        log "Found SATA root: $ROOTDEV"
        break
    fi
#    log "Waiting for SATA root ($i/$TIMEOUT)..."

    # Overwrite the current line with the new countdown
    printf '\r%s [root-switch] Waiting for SATA root (%2d/%d)...' \
           "$(date '+%Y-%m-%d %H:%M:%S')" "$i" "$TIMEOUT" \
           > /dev/serial0

    # Optional: also send to /dev/console in case it's different
    printf '\r%s [root-switch] Waiting for SATA root (%2d/%d)...' \
           "$(date '+%Y-%m-%d %H:%M:%S')" "$i" "$TIMEOUT" \
           > /dev/console

    sleep 1
done

if [ -z "$ROOTDEV" ]; then
    log "SATA root not found, staying on SD root"
    exit 0
fi

### MOUNT AND SWITCH ROOT ###
log "Mounting SATA root at $NEWROOT"
mkdir -p "$NEWROOT"
mount "$ROOTDEV" "$NEWROOT"

log "Switching root to $ROOTDEV"
exec systemctl switch-root "$NEWROOT" /sbin/init

######End Of File###^Z

# make script executable
sudo chmod +x /usr/local/sbin/root-switch.sh

sudo nano /etc/systemd/system/root-switch.service
[Unit]
DefaultDependencies=no
After=local-fs.target
Before=sysinit.target

[Service]
Type=oneshot
ExecStart=/usr/local/sbin/root-switch.sh

[Install]
WantedBy=sysinit.target
###End Of File###^Z

# create script to list slot,device,serial of md1
nano ./lsmdblk
#!/bin/bash

# Check if exactly one argument is provided
if [ $# -ne 1 ]; then
    echo "Usage: $0 <md-device>"
    echo "Example:"
    echo "  $0 /dev/md0"
    echo "  $0 /dev/md127"
    exit 1
fi

MD_DEVICE="$1"

# Basic validation: should look like /dev/md followed by number
if [[ ! "$MD_DEVICE" =~ ^/dev/md[0-9]+$ ]]; then
    echo "Error: Argument must be a valid md device (e.g. /dev/md0, /dev/md127)"
    exit 1
fi

# Check if the device actually exists
if [ ! -e "$MD_DEVICE" ]; then
    echo "Error: Device $MD_DEVICE does not exist."
    exit 1
fi

# ANSI color codes
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
RED='\033[0;31m'
GRAY='\033[0;90m'
NC='\033[0m'

printf "Slot  Device     Serial             State\n"
printf -- "----  ---------  -----------------  --------------------------------\n"

sudo mdadm --detail "$MD_DEVICE" | \
grep -E '^ *[0-9]+[[:space:]]+[0-9]+[[:space:]]+[0-9]+[[:space:]]+.* /dev/sd' | \
awk '
  {
    slot = $4;
    dev  = $NF;
    # State starts from field 5 to (NF-1)
    state = "";
    for (i=5; i<NF; i++) {
      state = state $i " ";
    }
    state = gensub(/[[:space:]]+$/, "", "g", state);  # trim trailing space
    printf "%s|%s|%s\n", slot, dev, state;
  }' | \
sort -t'|' -k1 -n | \
while IFS='|' read -r slot dev full_state; do
    serial=$(sudo smartctl -i "$dev" 2>/dev/null | awk '/^Serial Number:/ {print $3}' || echo "-")

    # Color logic
    color="$GRAY"
    if [[ "$full_state" == *"active sync"* ]]; then
        color="$GREEN"
    elif [[ "$full_state" == *"rebuilding"* || "$full_state" == *"resync"* || "$full_state" == *"recovery"* ]]; then
        color="$YELLOW"
    elif [[ "$full_state" == *"spare"* ]]; then
        color="$CYAN"
    elif [[ "$full_state" == *"faulty"* || "$full_state" == *"failed"* || "$full_state" == *"removed"* ]]; then
        color="$RED"
    fi

    printf "%4s  %-9s  %-17s  ${color}%s${NC}\n" \
        "$slot" "$dev" "${serial:--}" "$full_state"
done
^S^X
