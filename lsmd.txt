#!/bin/bash

# ANSI color codes
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
RED='\033[0;31m'
GRAY='\033[0;90m'
NC='\033[0m'

# Get active md devices
MD_DEVICES=$(awk '/^md[0-9]+/ {print "/dev/"$1}' /proc/mdstat)

if [ -z "$MD_DEVICES" ]; then
    echo "No md devices found."
    exit 1
fi

for MD_DEVICE in $MD_DEVICES; do
    echo
    echo "=== $MD_DEVICE ==="

    printf "Slot  Device     Serial             Health   State\n"
    printf -- "----  ---------  -----------------  -------  --------------------------------\n"

    sudo mdadm --detail "$MD_DEVICE" | \
    grep -E '^ *[0-9]+[[:space:]]+[0-9]+[[:space:]]+[0-9]+[[:space:]]+.* /dev/sd' | \
    awk '
      {
        slot = $4;
        dev  = $NF;

        state = "";
        for (i=5; i<NF; i++) {
          state = state $i " ";
        }

        sub(/[[:space:]]+$/, "", state);

        printf "%s|%s|%s\n", slot, dev, state;
      }' | \
    sort -t'|' -k1 -n | \
    while IFS='|' read -r slot dev full_state; do

        # Get serial
        serial=$(sudo smartctl -i "$dev" 2>/dev/null | \
            awk -F: '/^Serial Number:/ {gsub(/^[ \t]+/, "", $2); print $2}')

        # Get SMART health
        health_raw=$(sudo smartctl -H "$dev" 2>/dev/null | \
            awk -F: '/SMART overall-health self-assessment test result/ {gsub(/^[ \t]+/, "", $2); print $2}')

        # Fallback if unavailable
        serial=${serial:--}
        health_raw=${health_raw:-UNKNOWN}

        # Color for health
        health_color="$GRAY"
        if [[ "$health_raw" == *"PASSED"* ]]; then
            health_color="$GREEN"
        elif [[ "$health_raw" == *"FAILED"* ]]; then
            health_color="$RED"
        else
            health_color="$YELLOW"
        fi

        # Color for RAID state
        state_color="$GRAY"
        if [[ "$full_state" == *"active sync"* ]]; then
            state_color="$GREEN"
        elif [[ "$full_state" == *"rebuilding"* || "$full_state" == *"resync"* || "$full_state" == *"recovery"* ]]; then
            state_color="$YELLOW"
        elif [[ "$full_state" == *"spare"* ]]; then
            state_color="$CYAN"
        elif [[ "$full_state" == *"faulty"* || "$full_state" == *"failed"* || "$full_state" == *"removed"* ]]; then
            state_color="$RED"
        fi

        printf "%4s  %-9s  %-17s  ${health_color}%-7s${NC}  ${state_color}%s${NC}\n" \
            "$slot" "$dev" "$serial" "$health_raw" "$full_state"
    done
done
